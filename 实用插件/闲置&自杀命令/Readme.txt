命令列表

!kill                                  自杀
!afk 或者 !spec                         观察者模式
!join                                  接管一个幸存者
!infected                              接管一个特感


控制台变量
// 启用或禁用加入特感队伍
l4d_nth_joininfected "1"

// 启用或禁用加入幸存者队伍
l4d_nth_joinsurvivors "1"

// 自杀指令权限  0 = 禁用, 1 = 所有人可用, 2 = 仅幸存者可用, 3 = 仅特感可用
l4d_nth_kill "1"

// 启用或禁用观察者模式
l4d_nth_spectate "1"

// 新玩家加入时是否有提示消息
l4d_nth_message "1"

// 提示消息多久重复一次
l4d_nth_timerrepeat "30"
